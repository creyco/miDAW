import { Component } from '@angular/core';
import { ClientesService } from '../../../../services/clientes.service';
import { VehiculoService } from '../../../../services/vehiculo.service';

@Component({
  selector: 'app-formEditVehiculo',
  templateUrl: './formEditVehiculo.component.html',

    '../../usuarios/usuarios.component.css'],
  standalone: true,
})
export class FormEditVehiculoComponent {

  constructor(protected vehiService: VehiculoService, protected cliService: ClientesService) { }


}

        @for (cliente of cliService.clientes; track cliente.id_cliente) {

          (change)="vehiService.insertImg($event)">

        <img [src]="vehiService.vehiculoEdit?.pathImage" [alt]="vehiService.vehiculoEdit?.marca_modelo">
        }
        @if (vehiService.imgVehiculo) {
        <img [src]="vehiService.imgVehiculo" [alt]="vehiService.imgVehiculo">
        }
        @if(!vehiService.vehiculoEdit?.pathImage && !vehiService.imgVehiculo) {
        <i class="fa-solid fa-car"></i>
        <span>Insertar Imagen</span>

    <div class="container-Btn-Modal">
